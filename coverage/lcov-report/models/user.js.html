function addQuotation (values){
	var length = values.length;
	var withQuotation = "";
 
	for (var i = 0; i < length-1; i++){
		withQuotation += "'" + values[i] + "',";
	}
 
	withQuotation += "'" + values[length-1] + "'";
	return withQuotation;
